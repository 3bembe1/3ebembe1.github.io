**Класи: User, Guest, Task, Timer, Settings, Session, Statistics, NotificationService, AuthService, AuthUser.**



```
@startuml

class User {
  - id : int
  - name : String
  - email : String
}

class Guest {
}

class AuthUser {
  + syncData() : void
}

User <|-- Guest
User <|-- AuthUser


class Task {
  - taskId : int
  - title : String
  - description : String
  - completed : boolean
  + create() : void
  + update() : void
  + delete() : void
}

class Timer {
  - duration : int
  - remainingTime : int
  - isRunning : boolean
  + start() : void
  + pause() : void
  + reset() : void
}

class Session {
  - sessionId : int
  - startTime : Date
  - endTime : Date
  + calculateDuration() : int
}

class Settings {
  - workDuration : int
  - breakDuration : int
  + updateSettings() : void
}

class Statistics {
  - weekData : List<Session>
  + getWeeklyStats() : List<Session>
}

class NotificationService {
  + sendSound() : void
  + sendBrowserNotification() : void
}

class AuthService {
  + loginWithGoogle() : boolean
}


AuthUser "1" -- "*" Task : manages           
AuthUser "1" -- "*" Session : has            

Timer "1" -- "1" Session : creates           
Session "*" -- "1" Statistics : stored in    

Settings "1" -- "1" Timer : configures    

Timer ..> NotificationService : triggers    
AuthUser ..> AuthService : uses              
@enduml
```
